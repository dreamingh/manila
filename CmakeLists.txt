#cmake_minium_required(version 2.8)project(tcp LANGUAGES CXX)set(CMAKE_BUILD_TYPE "Debug")set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall")set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)set(RootPath "${CMAKE_CURRENT_SOURCE_DIR}/.")list(APPEND SourceFiles     ${RootPath}/index.cpp )include_directories(${RootPath}/include)add_executable(tcp.exe ${SourceFiles})
